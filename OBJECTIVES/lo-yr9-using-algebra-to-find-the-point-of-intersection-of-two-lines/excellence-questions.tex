{\LARGE \textbf{Using algebra to find the point of intersection of two lines — Excellence}}\\[0.2em]
Solve each system algebraically. Give the point of intersection as \mbox{$(x,y)$}.

\begin{tasks}(3)
\task Solve \mbox{$2x+y=7$} and \mbox{$x-y=2$}.
\task Solve \mbox{$3x+y=17$} and \mbox{$y=2x-3$}.
\task Solve \mbox{$4x-y=5$} and \mbox{$x+2y=10$}.
\task Solve \mbox{$2x+3y=4$} and \mbox{$x-y=-5$}.
\task Solve \mbox{$5x+2y=18$} and \mbox{$y=x-3$}.
\task Solve \mbox{$3x-2y=11$} and \mbox{$x+y=1$}.
\task Solve \mbox{$2x-y=8$} and \mbox{$4x+y=7$}.
\task Solve \mbox{$3x+2y=1$} and \mbox{$x-2y=9$}.
\task Solve \mbox{$y=4x-7$} and \mbox{$2x+y=14$}.
\task Solve \mbox{$y=\tfrac{1}{2}x+4$} and \mbox{$y=-x+10$}.
\task Solve \mbox{$y=\tfrac{3}{2}x-1$} and \mbox{$y=-\tfrac{1}{2}x+7$}.
\task Solve \mbox{$y=\tfrac{2}{3}x+5$} and \mbox{$y=-\tfrac{1}{3}x+1$}.
\task A concert charges \mbox{$C=4t+18$}. A second concert charges \mbox{$C=6t+6$}. Find the intersection.
\task Two phone plans are \mbox{$y=0.4x+12$} and \mbox{$y=0.7x+3$}. Find the intersection and explain what it means.
\task A taxi model is \mbox{$y=3x+5$}. A shuttle model is \mbox{$y=2x+11$}. Find the intersection and the common cost.
\task Solve \mbox{$2x+y=13$} and \mbox{$y=9-x$}.
\end{tasks}
