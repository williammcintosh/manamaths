{\LARGE \textbf{Angles around a point --- Proficient}}
\begin{tasks}(3)
\task {Angles around a point are \mbox{$x$}, \mbox{$140^\circ$}, and \mbox{$105^\circ$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$2x$}, \mbox{$95^\circ$}, and \mbox{$145^\circ$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$x+20^\circ$}, \mbox{$110^\circ$}, and \mbox{$150^\circ$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$x$}, \mbox{$2x$}, \mbox{$70^\circ$}, and \mbox{$110^\circ$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$x+15^\circ$}, \mbox{$x$}, \mbox{$80^\circ$}, and \mbox{$95^\circ$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$3x$}, \mbox{$x$}, \mbox{$75^\circ$}, and \mbox{$85^\circ$}. Find \mbox{$x$} and all four angles.}
\task {Three angles around a point are in the ratio \mbox{$2:3:4$}. Find all three angles.}
\task {Four angles around a point are in the ratio \mbox{$1:2:2:3$}. Find all four angles.}
\task {Two angles around a point are equal. The third angle is \mbox{$96^\circ$} bigger than each of the equal angles. Find all three angles.}
\task {Angles around a point are \mbox{$x$}, \mbox{$x+30^\circ$}, \mbox{$x+50^\circ$}, and \mbox{$x+100^\circ$}. Find \mbox{$x$}.}
\task {A full turn is shared into five equal angles around a point. Find one angle and state whether it is acute, right, obtuse, or reflex.}
\task {A student writes \mbox{$120^\circ + 130^\circ + 115^\circ = 365^\circ$} and says the diagram is correct. Explain why it is impossible.}
\end{tasks}
