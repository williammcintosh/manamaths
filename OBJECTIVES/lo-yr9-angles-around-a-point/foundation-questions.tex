{\LARGE \textbf{Angles around a point --- Foundation}}
\begin{tasks}(3)
\task {Angles around a point are \mbox{$110^\circ$}, \mbox{$140^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$95^\circ$}, \mbox{$125^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$150^\circ$}, \mbox{$80^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$70^\circ$}, \mbox{$100^\circ$}, \mbox{$110^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$85^\circ$}, \mbox{$95^\circ$}, \mbox{$100^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$90^\circ$}, \mbox{$120^\circ$}, \mbox{$60^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$130^\circ$}, \mbox{$150^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Angles around a point are \mbox{$72^\circ$}, \mbox{$88^\circ$}, \mbox{$95^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\task {Three angles around a point are equal. Find each angle.}
\task {Four angles around a point are equal. Find each angle.}
\task {One angle around a point is \mbox{$145^\circ$}. The other two angles are equal. Find each equal angle.}
\task {A student says the angles around a point add to \mbox{$180^\circ$}. Write the correct total.}
\task {Angles around a point are \mbox{$98^\circ$}, \mbox{$121^\circ$}, and \mbox{$x$}. Is \mbox{$x$} acute, right, obtuse, or reflex? Find it.}
\task {Angles around a point are \mbox{$40^\circ$}, \mbox{$40^\circ$}, \mbox{$140^\circ$}, and \mbox{$x$}. Find \mbox{$x$}.}
\end{tasks}
