{\LARGE \textbf{Pythagorean triples — Foundation}}
\begin{tasks}(3)
\task Which one is a Pythagorean triple: \mbox{$(3,4,5)$} or \mbox{$(3,4,6)$}?
\task Which one is a Pythagorean triple: \mbox{$(5,12,13)$} or \mbox{$(5,12,14)$}?
\task Which one is a Pythagorean triple: \mbox{$(8,15,17)$} or \mbox{$(8,15,18)$}?
\task A right-angled triangle has short sides 3 cm and 4 cm. Write the full triple.
\task A right-angled triangle has short sides 5 m and 12 m. Write the full triple.
\task A right-angled triangle has short sides 8 mm and 15 mm. Write the full triple.
\task Complete the triple: \mbox{$(6,8,\ ?)$}.
\task Complete the triple: \mbox{$(9,12,\ ?)$}.
\task Complete the triple: \mbox{$(10,24,\ ?)$}.
\task Complete the triple: \mbox{$(7,24,\ ?)$}.
\task Is \mbox{$(9,12,15)$} a Pythagorean triple? Write yes or no.
\task Is \mbox{$(12,16,20)$} a Pythagorean triple? Write yes or no.
\task Is \mbox{$(6,8,9)$} a Pythagorean triple? Write yes or no.
\task If \mbox{$(3,4,5)$} is a Pythagorean triple, write another triple made by doubling it.
\end{tasks}
