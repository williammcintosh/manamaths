\newcommand{\barchart}[1]{%
\begin{tikzpicture}[baseline=(current bounding box.center), scale=0.78]
\draw[thick, ->] (0,0) -- (5.0,0);
\draw[thick, ->] (0,0) -- (0,3.8);
\def\sep{0.8}
\def\width{0.52}
\foreach \cat/\shown/\h [count=\i] in {#1} {
  \pgfmathsetmacro{\x}{\i*\sep}
  \draw[fill=teal!50, draw=black, thick] (\x-\width/2,0) rectangle (\x+\width/2,\h*0.08);
  \node[below, font=\scriptsize] at (\x, -0.12) {\cat};
  \node[above, font=\scriptsize] at (\x, \h*0.08+0.06) {\shown};
}
\end{tikzpicture}%
}

\newcommand{\histogram}[1]{%
\begin{tikzpicture}[baseline=(current bounding box.center), scale=0.78]
\draw[thick, ->] (0,0) -- (5.1,0);
\draw[thick, ->] (0,0) -- (0,3.8);
\def\sep{0.9}
\def\width{0.82}
\foreach \int/\val [count=\i] in {#1} {
  \pgfmathsetmacro{\x}{\i*\sep-\width/2}
  \draw[fill=orange!50, draw=black, thick] (\x,0) rectangle (\x+\width,\val*0.16);
  \node[below, font=\tiny] at (\x+\width/2, -0.12) {\int};
  \node[above, font=\scriptsize] at (\x+\width/2, \val*0.16+0.06) {\val};
}
\end{tikzpicture}%
}

{\LARGE \textbf{Interpreting bar graphs \& histograms --- Excellence}}
\begin{tasks}(3)
\task Distribution of test scores.\\
\histogram{0-20/2,21-40/6,41-60/12,61-80/9,81-100/5}\\
Estimate the median score and explain.

\task Sales before and after a promotion.\\
\barchart{Before/45/45,After/68/68}\\
Calculate the percentage increase.

\task Ages of survey participants.\\
\histogram{10-19/8,20-29/15,30-39/22,40-49/18,50-59/7}\\
What is the modal class? Estimate the mean age.

\task Monthly energy use in kWh.\\
\barchart{Jan/320/32,Feb/290/29,Mar/310/31,Apr/280/28,May/260/26,Jun/240/24}\\
Describe the trend and suggest a reason.

\task Average test scores for three classes.\\
\barchart{A/76/76,B/82/82,C/79/79}\\
If each class has 25 students, what is the overall average?

\task Annual rainfall in mm for five years.\\
\barchart{Y1/1200/120,Y2/980/98,Y3/1100/110,Y4/1050/105,Y5/1150/115}\\
Calculate the interquartile range.
\end{tasks}
