\newcommand{\barchart}[1]{%
\begin{tikzpicture}[baseline=(current bounding box.center), scale=0.78]
\draw[thick, ->] (0,0) -- (4.6,0);
\draw[thick, ->] (0,0) -- (0,3.6);
\def\sep{0.75}
\def\width{0.48}
\foreach \cat/\shown/\h [count=\i] in {#1} {
  \pgfmathsetmacro{\x}{\i*\sep}
  \draw[fill=teal!50, draw=black, thick] (\x-\width/2,0) rectangle (\x+\width/2,\h*0.14);
  \node[below, font=\scriptsize] at (\x, -0.12) {\cat};
  \node[above, font=\scriptsize] at (\x, \h*0.14+0.06) {\shown};
}
\end{tikzpicture}%
}

\newcommand{\histogram}[1]{%
\begin{tikzpicture}[baseline=(current bounding box.center), scale=0.78]
\draw[thick, ->] (0,0) -- (4.8,0);
\draw[thick, ->] (0,0) -- (0,3.6);
\def\sep{0.9}
\def\width{0.82}
\foreach \int/\val [count=\i] in {#1} {
  \pgfmathsetmacro{\x}{\i*\sep-\width/2}
  \draw[fill=orange!50, draw=black, thick] (\x,0) rectangle (\x+\width,\val*0.22);
  \node[below, font=\tiny] at (\x+\width/2, -0.12) {\int};
  \node[above, font=\scriptsize] at (\x+\width/2, \val*0.22+0.06) {\val};
}
\end{tikzpicture}%
}
