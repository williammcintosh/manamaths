{\LARGE \textbf{Solving measurement equations --- Foundation}}
\begin{tasks}(3)
\task A square has perimeter \mbox{$20$} cm. Solve \mbox{$4s=20$}.
\task A rectangle has perimeter \mbox{$18$} m and width \mbox{$4$} m. Solve \mbox{$2l+8=18$}.
\task A rectangle has area \mbox{$24$} cm\mbox{$^2$} and width \mbox{$6$} cm. Solve \mbox{$6l=24$}.
\task A triangle has area \mbox{$15$} cm\mbox{$^2$} and height \mbox{$5$} cm. Solve \mbox{$5b=30$}.
\task A cuboid has volume \mbox{$36$} cm\mbox{$^3$}, width \mbox{$3$} cm, and height \mbox{$4$} cm. Solve \mbox{$12l=36$}.
\task Distance is \mbox{$42$} km and speed is \mbox{$6$} km/h. Solve \mbox{$6t=42$}.
\task Capacity is \mbox{$2.4$} L shared equally into \mbox{$8$} bottles. Solve \mbox{$8c=2.4$}.
\task A fence uses \mbox{$30$} m of wire around a regular pentagon. Solve \mbox{$5s=30$}.
\task A rectangle has perimeter \mbox{$34$} cm and length \mbox{$9$} cm. Solve \mbox{$18+2w=34$}.
\task A prism has volume \mbox{$45$} cm\mbox{$^3$} and cross-sectional area \mbox{$9$} cm\mbox{$^2$}. Solve \mbox{$9d=45$}.
\task A runner travels \mbox{$28$} km in \mbox{$4$} h. Solve \mbox{$4v=28$}.
\task A square has area \mbox{$49$} cm\mbox{$^2$}. Solve \mbox{$s^2=49$}.
\task A jug holds \mbox{$1500$} mL. Solve \mbox{$1000L=1500$} for \mbox{$L$} in litres.
\task A rectangle has area \mbox{$56$} m\mbox{$^2$} and length \mbox{$8$} m. Solve \mbox{$8w=56$}.
\end{tasks}
