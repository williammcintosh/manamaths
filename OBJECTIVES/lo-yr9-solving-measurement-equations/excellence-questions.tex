{\LARGE \textbf{Solving measurement equations --- Excellence}}
\begin{tasks}(3)
\task A rectangle has perimeter \mbox{$54$} cm, length \mbox{$2x+1$} cm, and width \mbox{$x+4$} cm. Solve \mbox{$2(2x+1)+2(x+4)=54$}.
\task A rectangle has area \mbox{$96$} cm\mbox{$^2$}, length \mbox{$x+2$} cm, and width \mbox{$x-2$} cm. Solve \mbox{$(x+2)(x-2)=96$}.
\task A triangle has area \mbox{$45$} m\mbox{$^2$}, base \mbox{$x+1$} m, and height \mbox{$6$} m. Solve \mbox{$3(x+1)=45$}.
\task A trapezium has area \mbox{$54$} cm\mbox{$^2$}, height \mbox{$6$} cm, and parallel sides \mbox{$x$} cm and \mbox{$x+3$} cm. Solve \mbox{$6x+9=54$}.
\task A cuboid has volume \mbox{$144$} cm\mbox{$^3$}, width \mbox{$x$} cm, height \mbox{$4$} cm, and length \mbox{$x+2$} cm. Solve \mbox{$4x(x+2)=144$}.
\task A prism has volume \mbox{$150$} cm\mbox{$^3$}, cross-sectional area \mbox{$3x$} cm\mbox{$^2$}, and length \mbox{$5$} cm. Solve \mbox{$15x=150$}.
\task A circle has circumference \mbox{$43.96$} cm. Using \mbox{$\pi=3.14$}, solve \mbox{$6.28r=43.96$}.
\task A circle has area \mbox{$78.5$} cm\mbox{$^2$}. Using \mbox{$\pi=3.14$}, solve \mbox{$3.14r^2=78.5$}.
\task Distance is \mbox{$252$} km. The speed is \mbox{$x+6$} km/h and the time is \mbox{$3$} h. Solve \mbox{$3(x+6)=252$}.
\task A cyclist rides at \mbox{$12$} km/h for \mbox{$t$} hours and covers \mbox{$54$} km. Solve \mbox{$12t=54$}.
\task A tank starts with \mbox{$18$} L and fills at \mbox{$4$} L/min for \mbox{$m$} minutes. It ends with \mbox{$50$} L. Solve \mbox{$18+4m=50$}.
\task A rectangle has perimeter \mbox{$70$} m and width \mbox{$w$} m. Its length is \mbox{$w+9$} m. Solve \mbox{$2w+2(w+9)=70$}.
\task A square and an equilateral triangle have the same perimeter. The triangle side is \mbox{$12$} cm. Solve \mbox{$4s=36$} for the square side \mbox{$s$}.
\task A cone has volume \mbox{$100\pi$} cm\mbox{$^3$}, radius \mbox{$5$} cm, and height \mbox{$h$} cm. Solve \mbox{$25h=300$}.
\task A sphere has volume \mbox{$288\pi$} cm\mbox{$^3$}. Solve \mbox{$r^3=216$}.
\task A rectangle has area \mbox{$72$} cm\mbox{$^2$} and width \mbox{$x$} cm. Its length is \mbox{$x+3$} cm. Solve \mbox{$x(x+3)=72$}.
\end{tasks}
