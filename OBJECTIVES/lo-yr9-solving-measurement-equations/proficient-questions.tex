{\LARGE \textbf{Solving measurement equations --- Proficient}}
\begin{tasks}(3)
\task A rectangle has perimeter \mbox{$26$} cm and width \mbox{$5$} cm. Solve \mbox{$2l+10=26$}.
\task A rectangle has perimeter \mbox{$42$} m and length \mbox{$x+3$} m, width \mbox{$x$} m. Solve \mbox{$2(x+3)+2x=42$}.
\task A rectangle has area \mbox{$63$} cm\mbox{$^2$} and length \mbox{$x+2$} cm, width \mbox{$7$} cm. Solve \mbox{$7(x+2)=63$}.
\task A triangle has area \mbox{$24$} m\mbox{$^2$} and base \mbox{$8$} m. Solve \mbox{$8h=48$}.
\task A trapezium has area \mbox{$40$} cm\mbox{$^2$}, parallel sides \mbox{$6$} cm and \mbox{$10$} cm. Solve \mbox{$16h=80$}.
\task A prism has volume \mbox{$84$} cm\mbox{$^3$} and cross-sectional area \mbox{$12$} cm\mbox{$^2$}. Solve \mbox{$12d=84$}.
\task A cuboid has volume \mbox{$90$} cm\mbox{$^3$} with width \mbox{$3$} cm and height \mbox{$5$} cm. Solve \mbox{$15l=90$}.
\task Speed is \mbox{$72$} km/h and time is \mbox{$2.5$} h. Solve \mbox{$2.5d=180$} for the distance \mbox{$d$}.
\task A car travels \mbox{$198$} km in \mbox{$3$} h. Solve \mbox{$3v=198$}.
\task A bottle contains \mbox{$c$} litres. Six bottles hold \mbox{$9$} L altogether. Solve \mbox{$6c=9$}.
\task A square has area \mbox{$121$} cm\mbox{$^2$}. Solve \mbox{$s^2=121$}.
\task A circle has circumference \mbox{$31.4$} cm. Using \mbox{$C=2\pi r$} and \mbox{$\pi=3.14$}, solve \mbox{$6.28r=31.4$}.
\end{tasks}
