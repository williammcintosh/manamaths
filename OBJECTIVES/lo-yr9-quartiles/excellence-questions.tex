{\LARGE \textbf{Quartiles — Excellence}}
\begin{tasks}(3)
\task For 4, 6, 8, 10, 12, 14, 30, find \mbox{$Q_1$}, \mbox{$Q_2$}, \mbox{$Q_3$}, and the IQR.
\task For 4, 6, 8, 10, 12, 14, 30, is the IQR or the range a better measure of spread? Why?
\task Set A: \mbox{$Q_1=14$}, \mbox{$Q_3=26$}. Set B: \mbox{$Q_1=18$}, \mbox{$Q_3=22$}. Both have median 20. Which set is more consistent?
\task Create 7 ordered numbers with \mbox{$Q_1=8$}, median 12, and \mbox{$Q_3=18$}.
\task Create 8 ordered numbers with \mbox{$Q_1=15$} and \mbox{$Q_3=27$}. Then find the IQR.
\task A student says, "IQR 12 means 12 of the data is in the middle." What is wrong?
\task Box plot: min 5, \mbox{$Q_1=11$}, median 16, \mbox{$Q_3=24$}, max 40. Find the IQR and the range.
\task For min 5, \mbox{$Q_1=11$}, median 16, \mbox{$Q_3=24$}, max 40, which measure is affected most by the maximum?
\task Set A summary: 2, 8, 12, 20, 22. Set B summary: 2, 10, 12, 14, 22. Which set has the larger IQR?
\task If \mbox{$Q_1=14$} and the IQR is 9, find \mbox{$Q_3$}.
\task If \mbox{$Q_1=18$} and \mbox{$Q_3=30$}, what part of the data lies in the box?
\task If \mbox{$Q_1=18$} and \mbox{$Q_3=30$}, what part lies between the median and \mbox{$Q_3$}?
\task For 7, 9, 10, 12, 15, 18, 21, 24, 30, find \mbox{$Q_1$}, \mbox{$Q_2$}, and \mbox{$Q_3$}.
\task For 7, 9, 10, 12, 15, 18, 21, 24, 30, which quarter has the greatest spread?
\task Two sets both have IQR 10. Set A has range 18. Set B has range 35. Compare them.
\task Write a rule for finding the IQR from a box plot.
\task Use your rule on: min 6, \mbox{$Q_1=13$}, median 17, \mbox{$Q_3=21$}, max 28.
\task Can the median lie outside the box? Explain.
\end{tasks}
