{\LARGE \textbf{Re‑arranging formula — Excellence}}
\begin{tasks}(3)
\task Given \mbox{$y = \dfrac{ax+b}{cx+d}$}, make \mbox{$x$} the subject.
\task The formula for the sum of an infinite geometric series is \mbox{$S = \dfrac{a}{1-r}$}. Make \mbox{$r$} the subject.
\task Given \mbox{$T = 2\pi\sqrt{\dfrac{m}{k}}$}, make \mbox{$k$} the subject.
\task The formula for the focal length of a lens is \mbox{$\dfrac{1}{f} = \dfrac{1}{u} + \dfrac{1}{v}$}. Make \mbox{$v$} the subject.
\task Given \mbox{$A = P\left(1 + \dfrac{r}{100}\right)^n$}, make \mbox{$r$} the subject.
\task The formula for the combined resistance of two resistors in parallel is \mbox{$\dfrac{1}{R} = \dfrac{1}{R_1} + \dfrac{1}{R_2}$}. Make \mbox{$R_1$} the subject.
\task Given \mbox{$x = \dfrac{-b \pm \sqrt{b^2 - 4ac}}{2a}$}, make \mbox{$b$} the subject.
\task The formula for the escape velocity is \mbox{$v = \sqrt{\dfrac{2GM}{r}}$}. Make \mbox{$r$} the subject.
\task Given \mbox{$y = \dfrac{1}{2}gt^2$}, make \mbox{$t$} the subject.
\task The formula for the magnification of a lens is \mbox{$M = \dfrac{v}{u}$}. Make \mbox{$u$} the subject.
\task Given \mbox{$PV = nRT$}, make \mbox{$T$} the subject.
\task The formula for the period of a mass‑spring system is \mbox{$T = 2\pi\sqrt{\dfrac{m}{k}}$}. Make \mbox{$m$} the subject.
\task Given \mbox{$E = \dfrac{hc}{\lambda}$}, make \mbox{$\lambda$} the subject.
\task The formula for the thin lens equation is \mbox{$\dfrac{1}{f} = \dfrac{1}{u} + \dfrac{1}{v}$}. Make \mbox{$u$} the subject.
\task Given \mbox{$S = \dfrac{n}{2}(2a + (n-1)d)$}, make \mbox{$d$} the subject.
\end{tasks}
