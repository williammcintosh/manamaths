{\LARGE \textbf{Re‑arranging formula — Foundation}}
\begin{tasks}(3)
\task The formula for distance is \mbox{$D = S \times T$}. Make \mbox{$T$} the subject.
\task The formula for perimeter of a rectangle is \mbox{$P = 2L + 2W$}. Make \mbox{$L$} the subject.
\task Given \mbox{$y = x + 5$}, re‑arrange the formula to make \mbox{$x$} the subject.
\task The formula for area of a rectangle is \mbox{$A = L \times W$}. Make \mbox{$W$} the subject.
\task The formula for speed is \mbox{$S = \dfrac{D}{T}$}. Make \mbox{$D$} the subject.
\task Given \mbox{$C = 2\pi r$}, make \mbox{$r$} the subject.
\task The formula for the area of a triangle is \mbox{$A = \dfrac{1}{2}bh$}. Make \mbox{$b$} the subject.
\task Given \mbox{$F = ma$}, make \mbox{$m$} the subject.
\task The formula for converting Celsius to Fahrenheit is \mbox{$F = \dfrac{9}{5}C + 32$}. Make \mbox{$C$} the subject.
\task Given \mbox{$V = IR$}, make \mbox{$I$} the subject.
\task The formula for density is \mbox{$D = \dfrac{M}{V}$}. Make \mbox{$M$} the subject.
\task Given \mbox{$E = mc^2$}, make \mbox{$m$} the subject.
\task The formula for simple interest is \mbox{$I = PRT$}. Make \mbox{$P$} the subject.
\task Given \mbox{$A = \pi r^2$}, make \mbox{$r$} the subject.
\task The formula for the volume of a cylinder is \mbox{$V = \pi r^2 h$}. Make \mbox{$h$} the subject.
\end{tasks}
