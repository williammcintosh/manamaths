{\LARGE \textbf{Interior angles of a triangle --- Excellence}}
\begin{tasks}(3)
\task {A triangle has angles \mbox{$(2x+8)^\circ$}, \mbox{$(x+16)^\circ$}, and \mbox{$60^\circ$}. Find \mbox{$x$}.}
\task {A triangle has angles \mbox{$(3y-4)^\circ$}, \mbox{$74^\circ$}, and \mbox{$(y+10)^\circ$}. Find \mbox{$y$}.}
\task {A triangle has angles \mbox{$(a+20)^\circ$}, \mbox{$a^\circ$}, and \mbox{$(2a-15)^\circ$}. Find \mbox{$a$}.}
\task {A triangle has angles \mbox{$b^\circ$}, \mbox{$b^\circ$}, and \mbox{$(b+12)^\circ$}. Find \mbox{$b$}.}
\task {An isosceles triangle has a vertex angle of \mbox{$(3m+12)^\circ$} and each base angle is \mbox{$(m+18)^\circ$}. Find \mbox{$m$}.}
\task {A triangle has angles in the ratio \mbox{$3:4:8$}. Find the angles and decide whether the triangle is acute, right, or obtuse.}
\task {Two angles in a triangle are equal. The third angle is \mbox{$26^\circ$} less than one of the equal angles. Find all three angles.}
\task {A triangle has angles \mbox{$(x+5)^\circ$}, \mbox{$(2x-15)^\circ$}, and \mbox{$(3x+10)^\circ$}. Find \mbox{$x$}.}
\task {A student says an isosceles triangle can have angles \mbox{$50^\circ$}, \mbox{$50^\circ$}, and \mbox{$90^\circ$}. Explain the mistake.}
\task {Can a triangle have two obtuse interior angles? Explain using the angle-sum rule.}
\task {A right-angled triangle has one acute angle double the other. Find the two acute angles.}
\task {An isosceles triangle has equal base angles. The vertex angle is one third of the sum of the two base angles. Find all three angles.}
\task {One angle of a triangle is \mbox{$15^\circ$} greater than another, and the third angle is twice the smallest angle. Find all three angles.}
\task {A triangle has angles in the ratio \mbox{$5:5:8$}. Find the angles and name the triangle using angle language.}
\task {Write your own triangle-angle question with one algebraic angle and two numeric angles. Then solve it.}
\task {Explain why every equilateral triangle is also isosceles using its interior angles.}
\end{tasks}
