{\LARGE \textbf{Interior angles of a triangle --- Proficient}}
\begin{tasks}(3)
\task {A triangle has angles \mbox{$(x+10)^\circ$}, \mbox{$60^\circ$}, and \mbox{$70^\circ$}. Find \mbox{$x$}.}
\task {A triangle has angles \mbox{$46^\circ$}, \mbox{$(y+18)^\circ$}, and \mbox{$66^\circ$}. Find \mbox{$y$}.}
\task {A triangle has angles \mbox{$2a^\circ$}, \mbox{$80^\circ$}, and \mbox{$60^\circ$}. Find \mbox{$a$}.}
\task {A triangle has angles \mbox{$52^\circ$}, \mbox{$b^\circ$}, and \mbox{$(b+14)^\circ$}. Find \mbox{$b$}.}
\task {A triangle has angles \mbox{$(c+9)^\circ$}, \mbox{$54^\circ$}, and \mbox{$c^\circ$}. Find \mbox{$c$}.}
\task {A triangle has angles \mbox{$d^\circ$}, \mbox{$d^\circ$}, and \mbox{$64^\circ$}. Find \mbox{$d$}.}
\task {An isosceles triangle has equal base angles. The top angle is \mbox{$44^\circ$}. Find each base angle.}
\task {An isosceles triangle has equal sides and one base angle of \mbox{$68^\circ$}. Find the other two angles.}
\task {A triangle has angles in the ratio \mbox{$2:3:4$}. Find the three angles.}
\task {A triangle has angles in the ratio \mbox{$1:1:2$}. Is it possible? Explain briefly.}
\task {A right-angled triangle has one acute angle of \mbox{$37^\circ$}. Find the other acute angle.}
\task {One angle in a triangle is twice another. The third angle is \mbox{$45^\circ$}. Find all three angles if the two related angles are consecutive multiples of the same amount.}
\end{tasks}
