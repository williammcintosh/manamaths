\task {A triangle has angles \mbox{$48^\circ$}, \mbox{$x$}, and \mbox{$61^\circ$}. Find \mbox{$x$}.}
\task {A triangle has angles \mbox{$72^\circ$}, \mbox{$x$}, and \mbox{$38^\circ$}. Find \mbox{$x$}.}
\task {A triangle has angles \mbox{$x$}, \mbox{$44^\circ$}, and \mbox{$50^\circ$}. Find \mbox{$x$}.}
