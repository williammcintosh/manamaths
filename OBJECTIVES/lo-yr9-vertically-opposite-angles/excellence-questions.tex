{\LARGE \textbf{Vertically opposite angles --- Excellence}}
\begin{tasks}(3)
\task {Find \mbox{$x$} and all four angles if the two vertically opposite angles are \mbox{$(3x-6)^\circ$} and the two adjacent angles are \mbox{$(2x+16)^\circ$}.}
\task {Find \mbox{$y$} and all four angles if one pair of vertically opposite angles is \mbox{$(5y-43)^\circ$} and the other pair is \mbox{$137^\circ$}.}
\task {Find \mbox{$a$} and all four angles if one pair of vertically opposite angles is \mbox{$(2a+9)^\circ$} and the other pair is \mbox{$(4a-3)^\circ$}.}
\task {Find \mbox{$b$}. Then state whether \mbox{$(3b+12)^\circ$} is acute or obtuse if its adjacent angle is \mbox{$(6b-18)^\circ$}.}
\task {Two vertically opposite angles are \mbox{$(7m-8)^\circ$} and \mbox{$(4m+22)^\circ$}. Find \mbox{$m$}.}
\task {One angle is \mbox{$(2n+5)^\circ$} and an adjacent angle is \mbox{$(5n-20)^\circ$}. Find \mbox{$n$}.}
\task {Find \mbox{$p$} if one angle is \mbox{$(p+31)^\circ$} and an adjacent angle is \mbox{$(2p-7)^\circ$}.}
\task {Find \mbox{$q$} if one angle is \mbox{$(4q+8)^\circ$} and an adjacent angle is \mbox{$(q+72)^\circ$}.}
\task {The obtuse angle in an intersecting-lines diagram is three times the acute angle. Find both angle sizes.}
\task {The acute angle is \mbox{$18^\circ$} less than the obtuse angle. Find both angle sizes.}
\task {A student measured one angle as about \mbox{$88^\circ$} and wrote the opposite angle as \mbox{$92^\circ$}. Explain what is wrong.}
\task {Create an expression for the angle opposite \mbox{$(x+27)^\circ$}. Then create an expression for an adjacent angle.}
\task {Find \mbox{$k$} if the acute angle is \mbox{$(k+14)^\circ$} and the obtuse angle is \mbox{$(2k+58)^\circ$}.}
\task {Explain why vertically opposite angles stay equal even if the intersecting lines are tilted or rotated.}
\task {If one angle in intersecting lines is doubled, does the vertically opposite angle also double? Explain.}
\task {Write one original vertically-opposite-angles problem of your own, then solve it.}
\end{tasks}
