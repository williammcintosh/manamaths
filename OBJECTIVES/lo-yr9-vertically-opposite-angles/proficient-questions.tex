{\LARGE \textbf{Vertically opposite angles --- Proficient}}
\begin{tasks}(3)
\task {Two vertically opposite angles are \mbox{$(x+14)^\circ$} and \mbox{$(3x-22)^\circ$}. Find \mbox{$x$}.}
\task {Two vertically opposite angles are \mbox{$(2y+5)^\circ$} and \mbox{$97^\circ$}. Find \mbox{$y$}.}
\task {Two vertically opposite angles are \mbox{$(2a-10)^\circ$} and \mbox{$(a+18)^\circ$}. Find \mbox{$a$}.}
\task {Two vertically opposite angles are \mbox{$(4b-9)^\circ$} and \mbox{$(2b+31)^\circ$}. Find \mbox{$b$}.}
\task {Two vertically opposite angles are \mbox{$(3c+12)^\circ$} and \mbox{$150^\circ$}. Find \mbox{$c$}.}
\task {Two vertically opposite angles are \mbox{$(5d-3)^\circ$} and \mbox{$(3d+39)^\circ$}. Find \mbox{$d$}.}
\task {One angle is \mbox{$x^\circ$}. The vertically opposite angle is \mbox{$(x+18)^\circ$}. What does this tell you about \mbox{$x$}?}
\task {Two vertically opposite angles are \mbox{$(2m+11)^\circ$} and \mbox{$(m+58)^\circ$}. Find \mbox{$m$}.}
\task {Two straight lines cross. One angle is \mbox{$58^\circ$}. Find the other three angles.}
\task {Two straight lines cross. One angle is \mbox{$133^\circ$}. Find the other three angles.}
\task {Explain why knowing one angle in an intersecting-lines diagram lets you find all four angles.}
\task {A student says angles of \mbox{$72^\circ$} and \mbox{$108^\circ$} are vertically opposite because they are across from each other in a sketch. What check should they make first?}
\end{tasks}
