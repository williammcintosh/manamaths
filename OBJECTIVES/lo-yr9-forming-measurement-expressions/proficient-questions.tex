{\LARGE \textbf{Forming measurement expressions --- Proficient}}
\begin{tasks}(3)
\task A rectangle has length \mbox{$x+3$} and width \mbox{$x$}. Write its perimeter.
\task A rectangle has length \mbox{$w+5$} and width \mbox{$w$}. Write its area.
\task A square has side \mbox{$y+2$}. Write its perimeter.
\task A path has \mbox{$n$} stones, each \mbox{$0.4$} m long. Write its total length.
\task One carton holds \mbox{$c$} L and another holds \mbox{$2$} L more. Write the total capacity.
\task A runner travels at \mbox{$v$} km/h for \mbox{$3$} hours. Write the distance.
\task A cuboid has length \mbox{$l$}, width \mbox{$w$}, height \mbox{$10$}. Write its volume.
\task A wire frame has \mbox{$6$} pieces of length \mbox{$p$} and one extra piece of \mbox{$12$}. Write the total length.
\task An equilateral triangle has side \mbox{$a+1$}. Write its perimeter.
\task A tank starts with \mbox{$40$} L and fills at \mbox{$r$} L/min for \mbox{$m$} minutes. Write the final amount.
\task A student writes the perimeter of sides \mbox{$x$} and \mbox{$x+4$} as \mbox{$2x+4$}. Write the correct expression.
\task A square has area \mbox{$s\times s$}. Write a simplified expression.
\end{tasks}
