{\LARGE \textbf{Understanding Cartesian plane and co-ordinates — Proficient}}
\begin{tasks}(3)
\task {Write the co-ordinates of point \mbox{$A$} if it is 4 right and 3 up. Then name the quadrant.}
\task {Write the co-ordinates of point \mbox{$B$} if it is 3 left and 4 up. Then name the quadrant.}
\task {Write the co-ordinates of point \mbox{$C$} if it is 2 left and 5 down. Then name the quadrant.}
\task {Write the co-ordinates of point \mbox{$D$} if it is 5 right and 1 down. Then name the quadrant.}
\task {Plot and label the points \mbox{$P(2,5)$}, \mbox{$Q(-4,2)$}, \mbox{$R(-1,-3)$} and \mbox{$S(3,-4)$}.}
\task {A point starts at \mbox{$(2,1)$}. It moves 3 units left and 4 units up. What are the new co-ordinates?}
\task {A point starts at \mbox{$(-5,3)$}. It moves 6 units right and 5 units down. What are the new co-ordinates?}
\task {If a point is on the \mbox{$x$}-axis, what must its \mbox{$y$}-value be?}
\task {If a point is on the \mbox{$y$}-axis, what must its \mbox{$x$}-value be?}
\task {Write these points in order from left to right: \mbox{$A(-4,1)$}, \mbox{$B(2,4)$}, \mbox{$C(3,-3)$}, \mbox{$D(-1,-2)$}.}
\task {Which point is exactly 4 units from the origin on the \mbox{$y$}-axis? Write both possibilities.}
\task {Which point is exactly 3 units from the origin on the \mbox{$x$}-axis? Write both possibilities.}
\task {Sam says \mbox{$(2,-5)$} means down 2 and right 5. Explain Sam's mistake.}
\task {Plot \mbox{$A(-3,0)$}, \mbox{$B(0,4)$} and \mbox{$C(3,0)$}. What are the co-ordinates of the vertex on the \mbox{$y$}-axis?}
\task {Write one point in each quadrant. Use integers only.}
\end{tasks}
