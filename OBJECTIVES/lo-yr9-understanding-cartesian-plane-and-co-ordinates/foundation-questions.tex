{\LARGE \textbf{Understanding Cartesian plane and co-ordinates — Foundation}}
\begin{tasks}(3)
\task {Write the co-ordinates of point \mbox{$A$} if you move 2 right and 3 up from the origin.}
\task {Write the co-ordinates of point \mbox{$B$} if you move 4 left and 2 up from the origin.}
\task {Write the co-ordinates of point \mbox{$C$} if you move 1 right and 3 down from the origin.}
\task {Plot the point \mbox{$P(3,2)$}.}
\task {Plot the point \mbox{$Q(-4,1)$}.}
\task {Plot the point \mbox{$R(0,-5)$}.}
\task {Plot the point \mbox{$S(-2,-3)$}.}
\task {Name the axis used first when writing a point in \mbox{$(x,y)$} form.}
\task {Name the axis used second when writing a point in \mbox{$(x,y)$} form.}
\task {A point is written as \mbox{$(4,-2)$}. How many units do you move in the \mbox{$y$} direction?}
\task {A point is written as \mbox{$(-3,5)$}. Do you move left or right first? Then up or down?}
\task {Write the co-ordinates of a point 4 units up on the \mbox{$y$}-axis.}
\task {Write the co-ordinates of a point 5 units left on the \mbox{$x$}-axis.}
\task {Write the co-ordinates of a point 2 units down on the \mbox{$y$}-axis.}
\task {Which point is 2 units to the right and 4 units up from the origin? Write it in \mbox{$(x,y)$} form.}
\task {Which point is 5 units to the left and 1 unit down from the origin? Write it in \mbox{$(x,y)$} form.}
\end{tasks}
