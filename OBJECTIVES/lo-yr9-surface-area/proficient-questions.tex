{\LARGE \textbf{Surface area — Proficient}}
\begin{tasks}(3)
\task A closed cuboid measures 12 cm by 8 cm by 5 cm. Find its surface area.
\task A storage box measures 1.2 m by 0.7 m by 0.5 m. Find its surface area.
\task A cube has surface area 384 cm\mbox{$^2$}. Find its side length.
\task A cuboid has dimensions 15 cm by 9 cm by 4 cm. Find its surface area.
\task A room is 5 m long, 4 m wide, and 2.4 m high. Find the area of the four walls only.
\task For the same room, find the area of the walls and ceiling, but not the floor.
\task A cylinder has radius 3 cm and height 10 cm. Using \mbox{$\pi \approx 3.14$}, find its total surface area.
\task A cylinder has diameter 8 m and height 12 m. Using \mbox{$\pi \approx 3.14$}, find its curved surface area only.
\task A cylinder has total surface area \mbox{$471$} cm\mbox{$^2$}. One circular end has area about \mbox{$78.5$} cm\mbox{$^2$}. Find the curved area.
\task A cuboid is 20 cm by 12 cm by 8 cm. How much paper is needed to cover all outside faces?
\task A cube has side length 0.4 m. Find its surface area in m\mbox{$^2$} and in cm\mbox{$^2$}.
\task A rectangular prism has surface area 214 cm\mbox{$^2$}. Its dimensions are 7 cm by 5 cm by \mbox{$h$} cm. Find \mbox{$h$}.
\end{tasks}
