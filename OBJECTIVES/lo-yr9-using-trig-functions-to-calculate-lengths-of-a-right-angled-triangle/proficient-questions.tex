{\LARGE \textbf{Using trig functions to calculate lengths of a right-angled triangle --- Proficient}}
\begin{tasks}(3)
\task Angle \mbox{$37^\circ$}, hypotenuse 16 cm. Find the adjacent side to 1 d.p.
\task Angle \mbox{$58^\circ$}, hypotenuse 24 m. Find the opposite side to 1 d.p.
\task Angle \mbox{$41^\circ$}, adjacent 9 cm. Find the hypotenuse to 1 d.p.
\task Angle \mbox{$73^\circ$}, opposite 12 m. Find the hypotenuse to 1 d.p.
\task Angle \mbox{$28^\circ$}, opposite 7 cm. Find the adjacent side to 1 d.p.
\task Angle \mbox{$64^\circ$}, adjacent 5 m. Find the opposite side to 1 d.p.
\task A kite string is 18 m long and makes an angle of \mbox{$52^\circ$} with the ground. Find the height to 1 d.p.
\task A tree shadow is 6.5 m long and the angle of elevation is \mbox{$49^\circ$}. Find the tree height to 1 d.p.
\task An 8 m ramp rises at \mbox{$9^\circ$}. Find the vertical rise to 2 d.p.
\task A 4.8 m roof beam makes an angle of \mbox{$35^\circ$} with the horizontal. Find the vertical drop to 1 d.p.
\task Angle of elevation to a building top is \mbox{$31^\circ$}. The point is 22 m from the base. Find the height to 1 d.p.
\task A guy wire is anchored 9 m from a pole and makes an angle of \mbox{$63^\circ$} with the ground. Find the wire length to 1 d.p.
