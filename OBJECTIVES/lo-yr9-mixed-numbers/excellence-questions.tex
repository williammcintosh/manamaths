\begin{tasks}(3)
\task A student says \mbox{$3\frac{4}{5}=\frac{19}{5}$}. Are they correct? Explain briefly.
\task A student says \mbox{$\frac{29}{6}=4\frac{4}{6}$}. Is this fully simplified? Explain.
\task Find \mbox{$2\frac{3}{4}+1\frac{5}{6}$}.
\task Find \mbox{$6\frac{2}{3}-2\frac{5}{12}$}.
\task Find \mbox{$3\frac{1}{2}+2\frac{2}{3}-1\frac{1}{6}$}.
\task Find \mbox{$7\frac{1}{4}-3\frac{5}{8}+1\frac{1}{2}$}.
\task Which is greater: \mbox{$4\frac{2}{3}$} or \mbox{$\frac{29}{6}$}? Justify your answer.
\task Complete both blanks: \mbox{$\frac{\square}{8}=5\frac{3}{8}=\frac{43}{\square}$}.
\task A baker used \mbox{$1\frac{3}{4}$} cups of flour in the morning and \mbox{$2\frac{2}{3}$} cups in the afternoon. How much flour was used altogether?
\task A runner completed \mbox{$9\frac{1}{2}$} km on Monday and \mbox{$3\frac{3}{4}$} km less on Tuesday. How far did the runner complete on Tuesday?
\task A ribbon that is \mbox{$5\frac{5}{6}$} m long is cut into pieces of length \mbox{$\frac{1}{6}$} m. How many pieces are made?
\task Write \mbox{$8\frac{7}{9}$} as an improper fraction, then say between which two whole numbers it lies.
\task Explain why \mbox{$\frac{17}{4}=4\frac{1}{4}$}.
\task Find a mixed number equal to \mbox{$\frac{52}{7}$}.
\task If \mbox{$x=2\frac{5}{6}$} and \mbox{$y=1\frac{1}{3}$}, find \mbox{$x+y$}.
\task If \mbox{$m=6\frac{1}{2}$} and \mbox{$n=2\frac{3}{5}$}, find \mbox{$m-n$}.
\end{tasks}
