{\LARGE \textbf{Using trig functions to calculate angles of a right-angled triangle --- Excellence}}
\begin{tasks}(3)
\task Drone: height \mbox{$48$} m, horizontal \mbox{$36$} m. Find the angle of elevation.
\task Guy wire: height \mbox{$14$} m, base \mbox{$5.5$} m. Find the ground angle.
\task Rescue ladder: length \mbox{$10.5$} m, rise \mbox{$9.2$} m. Find the angle.
\task Shed roof: rise \mbox{$1.8$} m, run \mbox{$3.6$} m. Find the pitch angle.
\task Cliff: base distance \mbox{$28$} m, angle \mbox{$63^\circ$}. Is the height closer to \mbox{$50$} m or \mbox{$60$} m? Justify.
\task Tree shadow \mbox{$9.4$} m, sun angle \mbox{$38.5^\circ$}. Find the tree height.
\task Ramp: length \mbox{$6.2$} m, rise \mbox{$0.65$} m. Does it stay below \mbox{$6^\circ$}?
\task Tent side: length \mbox{$2.8$} m, height \mbox{$2.1$} m. Find the ground angle.
\task Opposite \mbox{$13.7$} cm, adjacent \mbox{$13.7$} cm. Find the angle and explain.
\task Cable car: rise \mbox{$180$} m, slope \mbox{$420$} m. Find the angle.
\task Lighthouse: height \mbox{$22$} m, angle of depression \mbox{$34^\circ$}. Find the base distance.
\task Skate ramp: rise \mbox{$0.9$} m, run \mbox{$2.4$} m. Find the angle.
\task Roof beam: length \mbox{$5.6$} m, angle \mbox{$27^\circ$}. Find the rise.
\task Mountain path: rise \mbox{$320$} m, run \mbox{$1.05$} km. Find the angle.
\task Ladder angle \mbox{$72^\circ$}, base \mbox{$1.8$} m. Find the height reached.
\task Triangles with opposite:adjacent ratio \mbox{$3:4$} share what angle? Explain.
\end{tasks}
