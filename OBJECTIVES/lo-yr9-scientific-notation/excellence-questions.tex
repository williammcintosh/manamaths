\begin{tasks}(3)
\task {A number is written as \mbox{$a\times10^{n}$} where \mbox{$a$} has 3 significant figures. Explain how to convert it to 2 significant figures and give an example.}
\task {A calculation gives \mbox{$4.356\times10^{7}$}. Round this to 3 significant figures and explain the rounding step.}
\task {Compare \mbox{$9.99\times10^{2}$} and \mbox{$1.01\times10^{3}$}. Which is larger and by how much?}
\task {If \mbox{$5\times10^{x}=0.005$}, find \mbox{$x$} and show working.}
\task {Multiply \mbox{$(7.2\times10^{-4})\times(8.3\times10^{5})$} and give the answer in scientific notation, simplified.}
\task {A measurement is \mbox{$6.02\times10^{23}$}. Explain in one sentence what this means (student-facing).}
\task {Write \mbox{$0.00000000045$} in scientific notation and state the exponent and coefficient separately.}
\task {A spacecraft travels \mbox{$2.5\times10^{8}$} m in one hour. What is its speed in m/s in scientific notation?}
\task {Simplify: \mbox{$(3\times10^{4})\div(9\times10^{1})$} and express as scientific notation.}
\task {A value is given as \mbox{$4.00\times10^{-2}$} and another as \mbox{$4\times10^{-2}$}. Discuss any difference in significance.}
\task {Convert \mbox{$1.2345\times10^{3}$} to standard form and then to scientific notation with 2 sig figs.}
\task {Prove that multiplying by \mbox{$10^{n}$} shifts the decimal point by \mbox{$n$} places and give two examples.}
\end{tasks}
