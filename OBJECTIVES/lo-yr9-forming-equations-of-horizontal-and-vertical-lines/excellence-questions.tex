{\LARGE \textbf{Forming equations of horizontal and vertical lines — Excellence}}\\[0.2em]
Use the fixed coordinate and the gradient property.

\begin{tasks}(3)
\task {Horizontal line through \mbox{$(-9,11)$}. State its gradient.}
\task {Vertical line through \mbox{$(4,-8)$}. State its gradient property.}
\task {Which has gradient \mbox{$0$}: \mbox{$x=-3$} or \mbox{$y=-3$}?}
\task {Which has undefined gradient: \mbox{$x=6$} or \mbox{$y=6$}?}
\task {Through \mbox{$(2,5)$} with gradient \mbox{$0$}. Equation?}
\task {Through \mbox{$(-7,1)$} and vertical. Equation?}
\task {Through \mbox{$(3,-4)$}, parallel to \mbox{$y=10$}. Equation?}
\task {Through \mbox{$(8,2)$}, parallel to \mbox{$x=-1$}. Equation?}
\task {Points \mbox{$A(6,-3)$}, \mbox{$B(-2,-3)$}. Equation and gradient?}
\task {Points \mbox{$C(5,7)$}, \mbox{$D(5,-4)$}. Equation and gradient?}
\task {Rectangle \mbox{$(1,2),(7,2),(7,-3),(1,-3)$}. Equations of all sides?}
\task {Moves horizontally through \mbox{$(-4,9)$}. Fixed coordinate and equation?}
\task {Moves vertically through \mbox{$(12,-5)$}. Fixed coordinate and equation?}
\task {Do \mbox{$x=4$} and \mbox{$y=4$} have the same gradient?}
\task {How do all horizontal-line equations differ from vertical-line equations?}
\task {Compare the positions of \mbox{$y=-6$} and \mbox{$x=-6$}.}
\end{tasks}
