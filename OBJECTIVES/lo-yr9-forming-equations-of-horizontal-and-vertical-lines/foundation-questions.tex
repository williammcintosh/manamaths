{\LARGE \textbf{Forming equations of horizontal and vertical lines — Foundation}}\\[0.2em]
Horizontal lines use \mbox{$y=c$}. Vertical lines use \mbox{$x=c$}.

\begin{tasks}(3)
\task {Write the equation of the horizontal line through \mbox{$(4,3)$}.}
\task {Write the equation of the horizontal line through \mbox{$(-2,5)$}.}
\task {Write the equation of the horizontal line through \mbox{$(0,-4)$}.}
\task {Write the equation of the vertical line through \mbox{$(6,1)$}.}
\task {Write the equation of the vertical line through \mbox{$(-3,7)$}.}
\task {Write the equation of the vertical line through \mbox{$(0,-2)$}.}
\task {Is \mbox{$y=8$} horizontal or vertical?}
\task {Is \mbox{$x=-5$} horizontal or vertical?}
\task {Which equation matches a horizontal line: \mbox{$x=2$} or \mbox{$y=2$}?}
\task {Which equation matches a vertical line: \mbox{$x=-7$} or \mbox{$y=-7$}?}
\task {Point \mbox{$(9,-1)$} lies on a horizontal line. Write its equation.}
\task {Point \mbox{$(-4,6)$} lies on a vertical line. Write its equation.}
\task {Which line goes through \mbox{$(3,5)$} and \mbox{$(8,5)$}? Write the equation.}
\task {Which line goes through \mbox{$(-2,1)$} and \mbox{$(-2,9)$}? Write the equation.}
\end{tasks}
