{\LARGE \textbf{Representing and constructing 3D shapes --- Proficient}}
\begin{tasks}(3)
\task A solid has \mbox{$6$} faces, \mbox{$12$} edges, and \mbox{$8$} vertices. Name the solid.
\task A solid has \mbox{$5$} faces, \mbox{$9$} edges, and \mbox{$6$} vertices. Name the solid.
\task A solid has \mbox{$5$} faces, \mbox{$8$} edges, and \mbox{$5$} vertices. Name the solid.
\task Draw two different valid nets for a rectangular prism.
\task Draw a net for a triangular prism whose triangular ends are right triangles with sides \mbox{$3\text{ cm}$}, \mbox{$4\text{ cm}$}, and \mbox{$5\text{ cm}$}.
\task Draw a net for a square-based pyramid with base side \mbox{$6\text{ cm}$}.
\task A student draws a ``net'' for a rectangular prism using only \mbox{$5$} rectangles. Explain what is wrong.
\task A student draws a triangular prism net with only \mbox{$2$} rectangles. Explain what is missing.
\task Why does a row of \mbox{$6$} joined squares not fold to make a cube?
\task Describe the plan view of a rectangular prism with a \mbox{$7\text{ cm}$} by \mbox{$3\text{ cm}$} top face.
\task A square-based pyramid stands on its base. Describe its plan view and front view.
\task Explain one way to check whether a net will fold without faces overlapping.
\end{tasks}
