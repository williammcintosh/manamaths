{\LARGE \textbf{Determining nth term of pattern or spatial pattern — Foundation}}
\begin{tasks}(3)
\task Find the nth term of \mbox{$4,\ 7,\ 10,\ 13$}.
\task Find the nth term of \mbox{$9,\ 13,\ 17,\ 21$}.
\task Find the nth term of \mbox{$2,\ 6,\ 10,\ 14$}.
\task Find the nth term of \mbox{$15,\ 20,\ 25,\ 30$}.
\task The nth term is \mbox{$3n+1$}. Write term 6.
\task The nth term is \mbox{$4n-2$}. Write term 8.
\task The nth term is \mbox{$5n+3$}. Write term 10.
\task The nth term is \mbox{$2n+9$}. Which term equals \mbox{$25$}?
\task A tile pattern has totals \mbox{$5,\ 8,\ 11,\ 14$}. Find the nth term and then write term 12.
\task A growing row has totals \mbox{$7,\ 12,\ 17,\ 22$}. Find the nth term.
\task A border pattern has totals \mbox{$6,\ 10,\ 14,\ 18$}. Which term has \mbox{$34$} tiles?
\task A stair pattern has totals \mbox{$11,\ 15,\ 19,\ 23$}. Find the nth term and write term 9.
\task A student says the nth term of \mbox{$3,\ 7,\ 11,\ 15$} is \mbox{$4n$}. Is the student correct? If not, give the correct nth term.
\task A pattern follows \mbox{$8,\ 11,\ 14,\ 17$}. Does \mbox{$40$} appear in the pattern? Explain using the nth term.
\end{tasks}
