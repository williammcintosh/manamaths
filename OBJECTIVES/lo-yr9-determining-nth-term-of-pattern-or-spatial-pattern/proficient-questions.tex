{\LARGE \textbf{Determining nth term of pattern or spatial pattern — Proficient}}
\begin{tasks}(3)
\task Find the nth term of \mbox{$-2,\ 1,\ 4,\ 7,\ 10$}.
\task Find the nth term of \mbox{$14,\ 19,\ 24,\ 29$}.
\task Find the nth term of \mbox{$0.5,\ 2.5,\ 4.5,\ 6.5$}.
\task Find the nth term of \mbox{$18,\ 15,\ 12,\ 9$}.
\task If the nth term is \mbox{$6n-1$}, write term 15.
\task If the nth term is \mbox{$4n+7$}, which term is \mbox{$55$}?
\task Totals are \mbox{$7,\ 12,\ 17,\ 22$}. Find the nth term. Is \mbox{$60$} in the pattern?
\task A T-shape has totals \mbox{$4,\ 7,\ 10,\ 13$}. Find the nth term and term 20.
\task Matchsticks: \mbox{$6,\ 11,\ 16,\ 21$}. Which term has \mbox{$71$} matchsticks?
\task A student says \mbox{$5,\ 9,\ 13,\ 17$} has nth term \mbox{$n+4$}. Give the correct nth term.
\task Totals are \mbox{$9,\ 13,\ 17,\ 21$}. Find the nth term and the term number for \mbox{$81$}.
\task The 4th term is \mbox{$19$} and the 10th term is \mbox{$43$}. Find the nth term.
\end{tasks}
