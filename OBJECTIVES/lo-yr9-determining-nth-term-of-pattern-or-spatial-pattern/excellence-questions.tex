{\LARGE \textbf{Determining nth term of pattern or spatial pattern — Excellence}}
\begin{tasks}(3)
\task Find the nth term of \mbox{$3,\ 8,\ 13,\ 18,\ 23$}. Then write term 40.
\task Find the nth term of \mbox{$20,\ 16,\ 12,\ 8,\ 4$}. Is \mbox{$-28$} a term?
\task Find the nth term of \mbox{$1.5,\ 4.5,\ 7.5,\ 10.5$}. Then write term 25.
\task Totals are \mbox{$5,\ 9,\ 13,\ 17$}. Find the nth term and the term number for \mbox{$101$}.
\task Matchsticks: \mbox{$12,\ 17,\ 22,\ 27$}. Find the nth term and term 50.
\task Pattern \mbox{$4,\ 11,\ 18,\ 25$}: which term equals \mbox{$116$}?
\task A student gives \mbox{$7n+2$} for \mbox{$2,\ 9,\ 16,\ 23$}. Give the correct nth term.
\task If the nth term is \mbox{$8n-3$}, write three terms greater than \mbox{$100$}.
\task The 3rd term is \mbox{$17$} and the 9th term is \mbox{$47$}. Find the nth term.
\task The 5th term is \mbox{$27$} and the 12th term is \mbox{$55$}. Find the nth term.
\task Totals are \mbox{$10,\ 14,\ 18,\ 22$}. Give the nth term. Is term \mbox{$200$} possible?
\task If the nth term is \mbox{$5n+4$}, which term is \mbox{$149$}? What about \mbox{$150$}?
\task If the nth term is \mbox{$3n-8$}, find the first term greater than \mbox{$70$}.
\task The 1st term is \mbox{$6$} and the 8th term is \mbox{$34$}. Find the nth term.
\task Patterns \mbox{$4n+1$} and \mbox{$3n+9$}: find the first equal term.
\task Make a pattern with nth term \mbox{$6n-5$}. Write the first four terms and the term number for \mbox{$73$}.
\end{tasks}
