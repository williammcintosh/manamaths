{\LARGE \textbf{Naming angles --- Proficient}}
\begin{tasks}(3)
\task {Name the marked angle in two different ways.\\[0.4em]
\begin{tikzpicture}[scale=0.74,baseline=(current bounding box.center)]
  \coordinate (B) at (0,0); \coordinate (A) at (3.2,0.2); \coordinate (C) at (1.0,2.7);
  \draw[thick] (B)--(A); \draw[thick] (B)--(C);
  \draw[thick] (0.92,0.06) arc[start angle=4,end angle=70,radius=0.92];
  \fill (A) circle (1.7pt) node[right] {A}; \fill (B) circle (1.7pt) node[below left] {B}; \fill (C) circle (1.7pt) node[above] {C};
\end{tikzpicture}}
\task {Name the marked angle in two different ways.\\[0.4em]
\begin{tikzpicture}[scale=0.74,baseline=(current bounding box.center)]
  \coordinate (N) at (0,0); \coordinate (M) at (-0.2,2.8); \coordinate (P) at (3.0,0.1);
  \draw[thick] (N)--(M); \draw[thick] (N)--(P);
  \draw[thick] (0.88,0.04) arc[start angle=2,end angle=96,radius=0.88];
  \fill (M) circle (1.7pt) node[above] {M}; \fill (N) circle (1.7pt) node[below left] {N}; \fill (P) circle (1.7pt) node[right] {P};
\end{tikzpicture}}
\task {The rays \mbox{$QX$} and \mbox{$QY$} form an angle. Write two correct names for it.}
\task {The rays \mbox{$TS$} and \mbox{$TU$} form an angle. Write two correct names for it.}
\task {Which one is not a correct name for the same angle?\\[0.3em]\mbox{$\angle ABC$}, \mbox{$\angle CBA$}, \mbox{$\angle BAC$}}
\task {Which one is not a correct name for the same angle?\\[0.3em]\mbox{$\angle DEF$}, \mbox{$\angle FED$}, \mbox{$\angle DFE$}}
\task {Write the vertex of each angle: \mbox{$\angle JKL$}, \mbox{$\angle PQR$}, \mbox{$\angle STU$}.}
\task {Copy and fix the wrong name. The angle has vertex \mbox{$B$} and arms through \mbox{$A$} and \mbox{$C$}, but it was written as \mbox{$\angle ACB$}.}
\task {Copy and fix the wrong name. The angle has vertex \mbox{$M$} and arms through \mbox{$L$} and \mbox{$N$}, but it was written as \mbox{$\angle LNM$}.}
\task {At one point, three rays \mbox{$OA$}, \mbox{$OB$} and \mbox{$OC$} are drawn. Name the small angle between \mbox{$OA$} and \mbox{$OB$}.}
\task {At one point, three rays \mbox{$RV$}, \mbox{$RW$} and \mbox{$RX$} are drawn. Name the angle between \mbox{$RW$} and \mbox{$RX$}.}
\task {Explain why \mbox{$\angle ABC$} and \mbox{$\angle CBA$} can name the same angle, but \mbox{$\angle BAC$} cannot.}
\end{tasks}
