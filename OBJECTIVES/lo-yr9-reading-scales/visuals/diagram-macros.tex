\newcommand{\Thermometer}[2]{%
  \begin{tikzpicture}[scale=0.4,baseline=0]
    \node[above,font=\bfseries\small] at (0,5.5) {#1};
    \draw[thick] (0.15,-0.15)--(0.15,5);
    \foreach \y/\l in {0/-10,1/0,2/10,3/20,4/30,5/40}{
      \draw[thick] (0,\y)--(0.5,\y); \node[left,font=\tiny] at (-0.05,\y) {\l};}
    \node[above,font=\tiny] at (0.15,5) {50};
    \draw[very thick,rounded corners=1] (0.6,0) rectangle (1.1,5);
    \draw[very thick,rounded corners=2] (0.5,-0.25) rectangle (1.2,0.25);
    \fill[red!80!black] (0.6,0) rectangle (1.1,#2/10+1);
    \fill[red!80!black] (0.51,-0.24) rectangle (1.19,0.24);
    \draw[->,red,thick] (1.4,#2/10+1)--(1.8,#2/10+1);
    \node[right,red,font=\tiny] at (1.8,#2/10+1) {#2$^\circ$};
  \end{tikzpicture}%
}

\newcommand{\Ruler}[2]{%
  \begin{tikzpicture}[scale=0.4,baseline=0]
    \node[above,font=\bfseries\small] at (7.5,0.9) {#1};
    \draw[very thick] (0,0)--(15,0);
    \foreach \x in {0,...,15}{
      \draw[thick] (\x,0)--(\x,-0.15);
      \node[below,font=\tiny] at (\x,-0.17) {\x};}
    \foreach \x in {0.1,0.2,...,14.9}{\draw[thin] (\x,0)--(\x,-0.06);}
    \draw[->,red,thick] (#2,-0.35)--(#2,-0.15);
    \node[red,below,font=\tiny] at (#2,-0.38) {#2\,cm};
  \end{tikzpicture}%
}

\newcommand{\MeasuringJug}[2]{%

    \node[above,font=\bfseries\small] at (2,5.5) {#1};
    \draw[very thick] (0.4,0)--(0.4,5)--(1.0,5)--(1.6,5)--(2.2,5)--(2.2,0)--cycle;
    \foreach \y/\l in {0/0,1/200,2/400,3/600,4/800,5/1000}{
      \draw[thick] (2.2,\y)--(2.5,\y); \node[right,font=\tiny] at (2.6,\y) {\l};}
    \fill[blue!40!white,opacity=0.6] (0.41,0) rectangle (2.19,#2/200);

  \end{tikzpicture}%
}
