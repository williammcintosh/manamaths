{\LARGE \textbf{Transformation (Translation, reflection, enlargement) — Proficient}}\\[0.15em]
{\large Give full descriptions using vectors, mirror lines, or scale factors.}
\begin{tasks}(3)
\task Triangle $A$ has vertices $(1,1),(3,1),(2,4)$. Translate it by $\begin{pmatrix}4\\-2\end{pmatrix}$. Write the image coordinates.
\task Rectangle $B$ has vertices $(-3,-1),(-1,-1),(-1,2),(-3,2)$. Translate it by $\begin{pmatrix}2\\3\end{pmatrix}$. Write the image coordinates.
\task Point \mbox{$C(4,-1)$} is reflected in the \mbox{$y$}-axis. Write the image coordinates and name the mirror line.
\task Point \mbox{$D(-2,5)$} is reflected in the \mbox{$x$}-axis. Write the image coordinates and name the mirror line.
\task Shape \mbox{$E$} is enlarged from centre \mbox{$(0,0)$} by scale factor \mbox{$2$}. Find the image of \mbox{$(1,3)$}.
\task Shape \mbox{$F$} is enlarged from centre \mbox{$(0,0)$} by scale factor \mbox{$\tfrac12$}. Find the image of \mbox{$(-6,4)$}.
\task A side changes from \mbox{$5$} cm to \mbox{$15$} cm under an enlargement. Find the scale factor.
\task A side changes from \mbox{$12$} cm to \mbox{$3$} cm under an enlargement. Find the scale factor.
\task Describe fully the transformation from \mbox{$(2,1)$} to \mbox{$(2,-1)$}.
\task Describe fully the transformation from \mbox{$(-3,4)$} to \mbox{$(3,4)$}.
\task A point is reflected in the line \mbox{$x=2$}. The point \mbox{$(5,1)$} maps to what image?
\task A point is reflected in the line \mbox{$y=-1$}. The point \mbox{$(2,3)$} maps to what image?
\end{tasks}
