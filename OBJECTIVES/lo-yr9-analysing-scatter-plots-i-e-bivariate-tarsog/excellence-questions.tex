{\LARGE \textbf{Analysing scatter plots (i.e. bivariate) - TARSOG — Excellence}}
\begin{tasks}(3)
\task {The scatter plot shows the relationship between hours studied and test score (out of 100). Describe the relationship using TARSOG.\\[0.4em]
\begin{tikzpicture}[x=0.3cm,y=0.3cm,baseline=(current bounding box.center)]
  \draw[->] (0,0) -- (6,0) node[right] {Hours studied};
  \draw[->] (0,0) -- (0,12) node[above] {Test score};
  \foreach \x in {1,...,5} \draw (\x,0.2) -- (\x,-0.2) node[below] {\small \x};
  \foreach \y in {20,40,60,80,100} \draw (0.2,\y/10) -- (-0.2,\y/10) node[left] {\small \y};
  \fill[blue] (1,3) circle (0.15);
  \fill[blue] (2,5) circle (0.15);
  \fill[blue] (3,7) circle (0.15);
  \fill[blue] (4,8) circle (0.15);
  \fill[blue] (5,9) circle (0.15);
\end{tikzpicture}}
\task {The scatter plot shows the relationship between temperature (°C) and ice cream sales. Describe the relationship using TARSOG.\\[0.4em]
\begin{tikzpicture}[x=0.3cm,y=0.3cm,baseline=(current bounding box.center)]
  \draw[->] (0,0) -- (6,0) node[right] {Temperature (°C)};
  \draw[->] (0,0) -- (0,12) node[above] {Sales (hundreds)};
  \foreach \x in {10,20,30,40,50} \draw (\x/10,0.2) -- (\x/10,-0.2) node[below] {\small \x};
  \foreach \y in {2,4,6,8,10} \draw (0.2,\y) -- (-0.2,\y) node[left] {\small \y};
  \fill[blue] (1,2) circle (0.15);
  \fill[blue] (2,4) circle (0.15);
  \fill[blue] (3,6) circle (0.15);
  \fill[blue] (4,8) circle (0.15);
  \fill[blue] (5,10) circle (0.15);
\end{tikzpicture}}
\task {The scatter plot shows the relationship between age of a car and its resale value. Describe the relationship using TARSOG.\\[0.4em]
\begin{tikzpicture}[x=0.3cm,y=0.3cm,baseline=(current bounding box.center)]
  \draw[->] (0,0) -- (6,0) node[right] {Age (years)};
  \draw[->] (0,0) -- (0,12) node[above] {Value (thousands)};
  \foreach \x in {1,...,5} \draw (\x,0.2) -- (\x,-0.2) node[below] {\small \x};
  \foreach \y in {5,10,15,20,25} \draw (0.2,\y/2) -- (-0.2,\y/2) node[left] {\small \y};
  \fill[blue] (1,5) circle (0.15);
  \fill[blue] (2,4) circle (0.15);
  \fill[blue] (3,3) circle (0.15);
  \fill[blue] (4,2) circle (0.15);
  \fill[blue] (5,1) circle (0.15);
\end{tikzpicture}}
\task {Write a statistical paragraph for each of the three scatter plots above, using TARSOG to describe the relationship.}
\task {For the first scatter plot, predict the test score for a student who studied 6 hours. Explain your reasoning.}
\task {For the second scatter plot, predict ice cream sales on a 25°C day. Explain your reasoning.}
\task {For the third scatter plot, predict the resale value of a 6-year-old car. Explain your reasoning.}
\task {Identify any potential outliers in the scatter plots above and discuss how they might affect the interpretation.}
\task {Identify any groupings in the scatter plots above and suggest possible reasons for the groupings.}
\task {Discuss whether the relationships shown are likely to be causal or merely correlational.}
\task {Explain how you could collect data to create a scatter plot for a given bivariate relationship.}
\task {Explain the limitations of using scatter plots to analyse bivariate relationships.}
\task {Describe a real-world situation where a scatter plot would be useful, and what TARSOG aspects you would look for.}
\task {Compare and contrast scatter plots with other statistical displays (e.g., time series graphs, box plots).}
\task {Explain the importance of considering context when interpreting scatter plots.}
\task {Create a sketch of a scatter plot that shows a strong positive linear relationship with constant scatter and no outliers. Label axes appropriately.}
