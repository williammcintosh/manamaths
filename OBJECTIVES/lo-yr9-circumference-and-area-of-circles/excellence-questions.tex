{\LARGE \textbf{Circumference and area of circles — Excellence}}
\begin{tasks}(3)
\task {A circle's area is \mbox{$36\pi$} cm\mbox{$^2$}. Find its circumference in terms of \mbox{$\pi$}.}
\task {A circle's circumference is \mbox{$20\pi$} m. Find its area in terms of \mbox{$\pi$}.}
\task {The area of a circle is increased by 44\%. By what percentage does its radius increase?}
\task {The circumference of a circle is tripled. By what factor does its area increase?}
\task {A square and a circle have the same perimeter. If the square has side length 11 cm, find the area of the circle.}
\task {A rectangle has length 12 cm and width equal to the diameter of a circle with area \mbox{$25\pi$} cm\mbox{$^2$}. Find the rectangle's perimeter.}
\task {Find the area of the shaded region when a circle of radius 10 cm has a square of side length 14 cm inscribed in it.}
\task {Four semicircles of diameter 8 cm are arranged to form a shape. Find the total perimeter of the shape.}
\task {A circular sector has radius 15 cm and central angle \mbox{$72^\circ$}. Find its arc length and area.}
\task {Two concentric circles have radii 7 cm and 10 cm. Find the area of the annulus (ring) between them.}
\task {A wire is bent to form a circle of radius 21 cm. The same wire is then bent to form a square. Find the area of the square.}
\task {A circular flower bed has circumference 31.4 m. A path 1 m wide is built around it. Find the area of the path.}
\task {The ratio of the areas of two circles is 9:16. If the smaller circle has circumference \mbox{$18\pi$} cm, find the radius of the larger circle.}
\task {A running track consists of two straight sections each 100 m long and two semicircular ends. If the total length of the track is 400 m, find the area enclosed by the track.}
\task {A circle is inscribed in a square of side length 14 cm. Find the area of the region outside the circle but inside the square.}
\task {A goat is tied to a corner of a square shed with side length 10 m. The rope is 15 m long. Find the area the goat can graze (outside the shed).}
\task {Find the radius of a circle whose area is equal to the sum of the areas of two circles with radii 6 cm and 8 cm.}
\task {A circular pizza is cut into 8 equal slices. If the diameter is 40 cm, find the perimeter of one slice (including the two straight edges).}
\task {The minute hand of a clock is 12 cm long. Find the area swept by the minute hand in 25 minutes.}
\task {A circular pond has diameter 8 m. A concrete path 2 m wide is built around it. Find the cost of concreting the path at \$\mbox{$45$} per m\mbox{$^2$}.}
\end{tasks}
