{\LARGE \textbf{Circumference and area of circles — Proficient}}
\begin{tasks}(3)
\task {Find the circumference of a circle with radius 4.5 cm. Give your answer in terms of \mbox{$\pi$}.}
\task {Find the area of a circle with diameter 18 m. Give your answer in terms of \mbox{$\pi$}.}
\task {A circular garden has circumference 25.12 m. What is its radius? (Use \mbox{$\pi \approx 3.14$})}
\task {A circular pizza has area 154 cm\mbox{$^2$}. What is its diameter? (Use \mbox{$\pi \approx \frac{22}{7}$})}
\task {Which has larger area: a circle with radius 8 cm or a circle with circumference 50 cm?}
\task {A semicircle has diameter 20 cm. Find its perimeter (including the straight edge).}
\task {A quarter circle has radius 12 m. Find its area.}
\task {A bicycle wheel has radius 35 cm. How many complete revolutions does it make over 1 km?}
\task {Two circles have radii in the ratio 3:4. What is the ratio of their circumferences?}
\task {Two circles have radii in the ratio 2:5. What is the ratio of their areas?}
\task {A circular running track has inner radius 30 m and outer radius 35 m. Find the area of the track.}
\task {Find the perimeter of a shape made from a semicircle of radius 7 cm on top of a rectangle 14 cm by 7 cm.}
\end{tasks}
