{\LARGE \textbf{Regular/irregular polygon --- Proficient}}
\begin{tasks}(3)
\task {A polygon has all sides equal and all interior angles equal. Explain why it must be regular.}
\task {A polygon has six equal sides but its interior angles are not all equal. Is it regular or irregular? Explain briefly.}
\task {Find the sum of the interior angles of a heptagon.}
\task {Find the sum of the interior angles of a nonagon.}
\task {The sum of the interior angles of a polygon is \mbox{$1260^\circ$}. How many sides does the polygon have?}
\task {The sum of the interior angles of a polygon is \mbox{$1440^\circ$}. How many sides does the polygon have?}
\task {Find one interior angle of a regular decagon.}
\task {Find one interior angle of a regular nonagon.}
\task {An irregular pentagon has angles \mbox{$95^\circ$}, \mbox{$115^\circ$}, \mbox{$130^\circ$}, \mbox{$140^\circ$}, and \mbox{$x^\circ$}. Find \mbox{$x$}.}
\task {An irregular hexagon has angles \mbox{$x^\circ$}, \mbox{$125^\circ$}, \mbox{$135^\circ$}, \mbox{$145^\circ$}, \mbox{$155^\circ$}, and \mbox{$160^\circ$}. Find \mbox{$x$}.}
\task {A regular polygon has each interior angle equal to \mbox{$135^\circ$}. Name the polygon.}
\task {A regular polygon has each interior angle equal to \mbox{$150^\circ$}. How many sides does it have?}
\end{tasks}
