{\LARGE \textbf{Regular/irregular polygon --- Foundation}}
\begin{tasks}(3)
\task {State whether each polygon is regular or irregular: a pentagon with all sides equal and all angles equal.}
\task {State whether each polygon is regular or irregular: a hexagon with equal sides but one angle different from the others.}
\task {State whether each polygon is regular or irregular: a quadrilateral with angles \mbox{$90^\circ$}, \mbox{$90^\circ$}, \mbox{$90^\circ$}, and \mbox{$90^\circ$} but side lengths \mbox{\mbox{$3$} cm} and \mbox{\mbox{$5$} cm}.}
\task {Complete the sentence: in a regular polygon, all sides are the same length and all interior angles are equal.}
\task {Find the sum of the interior angles of a triangle.}
\task {Find the interior-angle sum of a quadrilateral.}
\task {Find the sum of the interior angles of a pentagon using \mbox{$180(n-2)$}.}
\task {Find the sum of the interior angles of a hexagon using \mbox{$180(n-2)$}.}
\task {Find the sum of the interior angles of an octagon using \mbox{$180(n-2)$}.}
\task {A regular pentagon has five equal interior angles. Find one interior angle.}
\task {A regular hexagon has six equal interior angles. Find one interior angle.}
\task {A regular octagon has eight equal interior angles. Find one interior angle.}
\task {An irregular pentagon has interior angles \mbox{$100^\circ$}, \mbox{$120^\circ$}, \mbox{$110^\circ$}, \mbox{$140^\circ$}, and \mbox{$x^\circ$}. Find \mbox{$x$}.}
\task {An irregular hexagon has interior angles \mbox{$120^\circ$}, \mbox{$140^\circ$}, \mbox{$130^\circ$}, \mbox{$150^\circ$}, \mbox{$110^\circ$}, and \mbox{$x^\circ$}. Find \mbox{$x$}.}
\end{tasks}
