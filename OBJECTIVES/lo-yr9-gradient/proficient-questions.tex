{\LARGE \textbf{Gradient — Proficient}}\\[0.2em]
Find gradients from points, tables, and patterns.

\begin{tasks}(4)
\task {Gradient from \mbox{$(1,2)$} to \mbox{$(5,4)$}.}
\task {Gradient from \mbox{$(-2,3)$} to \mbox{$(2,-3)$}.}
\task {Gradient from \mbox{$(4,-1)$} to \mbox{$(8,5)$}.}
\task {Gradient from \mbox{$(-3,-2)$} to \mbox{$(1,0)$}.}
\task {Rise \mbox{$3$}, run \mbox{$2$}. Gradient?}
\task {Fall \mbox{$5$}, run \mbox{$4$}. Gradient?}
\task {Rise \mbox{$7$}, run \mbox{$5$}. Gradient?}
\task {Fall \mbox{$9$}, run \mbox{$2$}. Gradient?}
\task {Table \mbox{$(0,-1),(2,2),(4,5),(6,8)$}. Gradient?}
\task {Table \mbox{$(-1,5),(1,2),(3,-1),(5,-4)$}. Gradient?}
\task {Pattern \mbox{$2,2.5,3,3.5,\dots$}. Change?}
\task {Pattern \mbox{$20,18.5,17,15.5,\dots$}. Change?}
\end{tasks}
