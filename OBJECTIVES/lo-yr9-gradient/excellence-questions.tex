{\LARGE \textbf{Gradient — Excellence}}\\[0.2em]
Work carefully with negative and fractional gradients.

\begin{tasks}(3)
\task {Gradient from \mbox{$(-3,-1)$} to \mbox{$(5,4)$}.}
\task {Gradient from \mbox{$(-4,5)$} to \mbox{$(2,-1)$}.}
\task {Gradient from \mbox{$(1,-2)$} to \mbox{$(9,3)$}.}
\task {Gradient from \mbox{$(-6,-4)$} to \mbox{$(2,6)$}.}
\task {Rise \mbox{$5$}, run \mbox{$2$}. Gradient?}
\task {Fall \mbox{$7$}, run \mbox{$3$}. Gradient?}
\task {Rise \mbox{$3$}, run \mbox{$4$}. Gradient?}
\task {Fall \mbox{$11$}, run \mbox{$5$}. Gradient?}
\task {Table \mbox{$(0,-3),(4,0),(8,3),(12,6)$}. Gradient?}
\task {Table \mbox{$(-2,7),(2,4),(6,1),(10,-2)$}. Gradient?}
\task {Table \mbox{$(1,2.5),(3,4),(5,5.5),(7,7)$}. Gradient?}
\task {Pattern \mbox{$18,13,8,3,\dots$}. Change?}
\task {Pattern \mbox{$1,1.75,2.5,3.25,\dots$}. Change?}
\task {Steeper: \mbox{$\frac{3}{4}$} or \mbox{$-2$}?}
\task {Which goes down faster: \mbox{$-\frac{5}{2}$} or \mbox{$\frac{1}{2}$}?}
\task {What does gradient \mbox{$0$} look like?}
\end{tasks}
