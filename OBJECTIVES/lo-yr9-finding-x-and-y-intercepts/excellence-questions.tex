{\LARGE \textbf{Finding x- and y-intercepts — Excellence}}\\[0.2em]
Work with equations, forms, and conditions.

\begin{tasks}(3)
\task {Find both intercepts of \mbox{$y=-\frac{2}{3}x+4$}.}
\task {Find both intercepts of \mbox{$y=\frac{5}{2}x-10$}.}
\task {Find both intercepts of \mbox{$4x+5y=20$}.}
\task {Find both intercepts of \mbox{$3x-4y=24$}.}
\task {Which line has x-intercept \mbox{$4$}: \mbox{$y=2x-8$} or \mbox{$y=-2x+8$}?}
\task {Which line has y-intercept \mbox{$-6$}: \mbox{$y=3x-6$} or \mbox{$y=-6x+3$}?}
\task {Intercepts \mbox{$(0,9)$} and \mbox{$(3,0)$}. Write the equation.}
\task {Intercepts \mbox{$(0,-8)$} and \mbox{$(-4,0)$}. Write the equation.}
\task {Find \mbox{$k$} if \mbox{$y=2x+k$} has x-intercept \mbox{$-3$}.}
\task {Find \mbox{$k$} if \mbox{$y=-4x+k$} has y-intercept \mbox{$(0,11)$}.}
\task {Through \mbox{$(0,5)$} and x-intercept \mbox{$(2,0)$}. Write the equation.}
\task {Through \mbox{$(0,-7)$} and x-intercept \mbox{$(-1,0)$}. Write the equation.}
\task {True or false: every non-horizontal line has one x- and one y-intercept.}
\task {Can \mbox{$y=6$} have an x-intercept? Briefly explain.}
\task {Can \mbox{$x=-3$} have a y-intercept? Briefly explain.}
\task {Larger x-intercept: \mbox{$y=-x+4$} or \mbox{$y=2x-4$}?}
\end{tasks}
