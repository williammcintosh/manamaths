{\LARGE \textbf{Finding x- and y-intercepts — Foundation}}\\[0.2em]
Use \mbox{$x=0$} for the y-intercept and \mbox{$y=0$} for the x-intercept.

\begin{tasks}(3)
\task {For \mbox{$y=x+4$}, find the y-intercept.}
\task {For \mbox{$y=x-3$}, find the y-intercept.}
\task {For \mbox{$y=2x+5$}, find the y-intercept.}
\task {For \mbox{$y=4x-7$}, find the y-intercept.}
\task {For \mbox{$y=-x+6$}, find the y-intercept.}
\task {For \mbox{$y=-3x-2$}, find the y-intercept.}
\task {For \mbox{$y=x+4$}, find the x-intercept.}
\task {For \mbox{$y=x-3$}, find the x-intercept.}
\task {For \mbox{$y=2x+8$}, find the x-intercept.}
\task {For \mbox{$y=3x-9$}, find the x-intercept.}
\task {For \mbox{$y=-x+6$}, find the x-intercept.}
\task {For \mbox{$y=-2x-10$}, find the x-intercept.}
\task {Line through \mbox{$(0,5)$} and \mbox{$(2,7)$}. What is the y-intercept?}
\task {Line through \mbox{$(-4,0)$} and \mbox{$(0,6)$}. Write both intercepts.}
\end{tasks}
