{\LARGE \textbf{Perimeter and area of composite shapes involving circles — Foundation}}
\begin{tasks}(3)
\task A rectangle is 8 cm by 4 cm with a semicircle of diameter 4 cm attached to one short side. Find the perimeter. Use \mbox{$\pi \approx 3.14$}.
\task A rectangle is 8 cm by 4 cm with a semicircle of diameter 4 cm attached to one short side. Find the area. Use \mbox{$\pi \approx 3.14$}.
\task A square of side 8 m has a quarter circle of radius 4 m cut from one corner. Find the remaining area. Use \mbox{$\pi \approx 3.14$}.
\task A rectangle is 10 m by 5 m with a semicircle of diameter 5 m attached along one long side. Find the perimeter. Use \mbox{$\pi \approx 3.14$}.
\task A shape is made from a rectangle 12 cm by 6 cm with a semicircle of diameter 6 cm attached to one short side. Find its area. Use \mbox{$\pi \approx 3.14$}.
\task A shape is made from a rectangle 12 cm by 6 cm with a semicircle of diameter 6 cm attached to one short side. Find its perimeter. Use \mbox{$\pi \approx 3.14$}.
\task A square of side 10 cm has a semicircle of diameter 10 cm attached along one side. Find the total perimeter. Use \mbox{$\pi \approx 3.14$}.
\task A square of side 10 cm has a semicircle of diameter 10 cm attached along one side. Find the total area. Use \mbox{$\pi \approx 3.14$}.
\task A quarter circle of radius 7 cm is cut from a rectangle 7 cm by 9 cm. Find the remaining area. Use \mbox{$\pi \approx 3.14$}.
\task A quarter circle of radius 7 cm is cut from a rectangle 7 cm by 9 cm. Find the perimeter of the remaining shape. Use \mbox{$\pi \approx 3.14$}.
\task A running lane has two straight sides of 20 m and two semicircular ends of diameter 8 m. Find the perimeter. Use \mbox{$\pi \approx 3.14$}.
\task A running lane has two straight sides of 20 m and two semicircular ends of diameter 8 m. Find the area inside the lane. Use \mbox{$\pi \approx 3.14$}.
\end{tasks}
