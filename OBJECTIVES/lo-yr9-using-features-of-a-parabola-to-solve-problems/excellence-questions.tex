{\LARGE \textbf{Using features of a parabola to solve problems — Excellence}}\\[0.2em]
{\large Use features of each parabola to justify your answers.}
\begin{tasks}(3)
\task For \mbox{$y=-(x-2)^2+9$}, find the maximum value, both roots, and the interval where \mbox{$y\ge 5$}.
\task For \mbox{$y=(x+1)^2-16$}, find the roots and the interval where \mbox{$y<0$}.
\task For \mbox{$y=-2(x-0)^2+8$}, compare its width with \mbox{$y=-x^2+8$}. Which is narrower, and why?
\task For \mbox{$y=\tfrac12(x-3)^2-2$}, estimate both roots to 1 decimal place and state the axis of symmetry.
\task For \mbox{$y=-(x+4)^2+1$}, determine whether the point \mbox{$(-6,-3)$} lies on the graph without expanding.
\task For \mbox{$y=(x-2)^2-9$}, find two different points with \mbox{$y=-5$}.
\task For \mbox{$y=-(x-1)^2+4$}, the graph represents profit. For which \mbox{$x$}-values is profit positive?
\task For \mbox{$y=(x+3)^2-4$}, one root is \mbox{$x=-1$}. Use symmetry to find the other root, then check by inspection.
\task For \mbox{$y=-\tfrac12(x-2)^2+6$}, estimate the interval where \mbox{$y>4$}.
\task For \mbox{$y=2(x+1)^2-8$}, find the roots and explain how the factor \mbox{$2$} changes the shape.
\task For \mbox{$y=-(x-0)^2+12$}, a jump lasts while \mbox{$y\ge 3$}. Find the horizontal interval for the jump.
\task For \mbox{$y=(x-4)^2+1$}, explain why there are no real roots.
\task For \mbox{$y=-(x+2)^2+9$}, find the \mbox{$y$}-intercept and use symmetry to find another point with the same height.
\task For \mbox{$y=(x-1)^2-4$}, find all \mbox{$x$}-values where the graph is at most \mbox{$5$}.
\end{tasks}
