{\LARGE \textbf{Using features of a parabola to solve problems — Foundation}}\\[0.2em]
{\large For each parabola, use the vertex, intercepts and symmetry to answer.}
\begin{tasks}(3)
\task For \mbox{$y=(x-0)^2-4$}, write the vertex and both \mbox{$x$}-intercepts.
\task For \mbox{$y=-(x-0)^2+9$}, write the vertex and the maximum value of \mbox{$y$}.
\task For \mbox{$y=(x-2)^2-1$}, write the axis of symmetry and the vertex.
\task For \mbox{$y=(x+3)^2-4$}, write both \mbox{$x$}-intercepts.
\task For \mbox{$y=-(x-1)^2+4$}, write both \mbox{$x$}-intercepts.
\task For \mbox{$y=(x-0)^2+5$}, write the vertex and say whether there are any \mbox{$x$}-intercepts.
\task For \mbox{$y=(x+1)^2-9$}, write the axis of symmetry and both \mbox{$x$}-intercepts.
\task For \mbox{$y=-(x+2)^2+1$}, write the vertex and the greatest value of \mbox{$y$}.
\task For \mbox{$y=(x-4)^2$}, write the vertex and the \mbox{$x$}-intercept.
\task For \mbox{$y=(x-1)^2-4$}, the graph models a ball path. When is the ball on the ground?
\task For \mbox{$y=-(x-3)^2+9$}, what is the highest point on the graph?
\task For \mbox{$y=(x+2)^2-1$}, one point is \mbox{$(-1,0)$}. Use symmetry to write the other \mbox{$x$}-intercept.
\end{tasks}
