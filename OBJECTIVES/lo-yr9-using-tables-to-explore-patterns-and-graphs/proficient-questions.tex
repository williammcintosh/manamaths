{\LARGE \textbf{Using tables to explore patterns and graphs — Proficient}}\\[0.2em]
Use each table to find a rule, extend the pattern, or predict what the graph would look like.

\begin{tasks}(3)
\task {Find the rule and write the next value of \mbox{$y$}.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $0$ & $1$ & $2$ & $3$ \\\hline
$y$ & $1$ & $4$ & $7$ & $10$ \\\hline
\end{tabular}}

\task {Find the rule and write the next value of \mbox{$y$}.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $1$ & $2$ & $3$ & $4$ \\\hline
$y$ & $8$ & $6$ & $4$ & $2$ \\\hline
\end{tabular}}

\task {Find the rule and write the next value of \mbox{$y$}.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $0$ & $2$ & $4$ & $6$ \\\hline
$y$ & $3$ & $7$ & $11$ & $15$ \\\hline
\end{tabular}}

\task {Complete the table for \mbox{$y=4x-1$}. Then write the ordered pair for \mbox{$x=5$}.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $1$ & $2$ & $3$ & $4$ \\\hline
$y$ &  &  &  &  \\\hline
\end{tabular}}

\task {Complete the table for \mbox{$y=12-3x$}. Then decide whether the graph rises or falls.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $0$ & $1$ & $2$ & $3$ \\\hline
$y$ &  &  &  &  \\\hline
\end{tabular}}

\task {Complete the table for \mbox{$y=0.5x+1$}. Then write the ordered pair for \mbox{$x=8$}.\\[0.3em]
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $0$ & $2$ & $4$ & $6$ \\\hline
$y$ &  &  &  &  \\\hline
\end{tabular}}

\task A table shows \mbox{$(0,2)$}, \mbox{$(1,5)$}, \mbox{$(2,8)$}, \mbox{$(3,11)$}. Write the rule and say whether the graph is a straight line or not.

\task A table shows \mbox{$(1,9)$}, \mbox{$(2,7)$}, \mbox{$(3,5)$}, \mbox{$(4,3)$}. Write the rule and predict the \mbox{$x$}-intercept.

\task A pattern has rule \mbox{$y=2x+3$}. Which is larger: the value when \mbox{$x=7$} or the value when \mbox{$x=10$}? By how much?

\task A pattern has rule \mbox{$y=15-x$}. Write three ordered pairs and describe the slope in words.

\task The table is \mbox{$(0,0)$}, \mbox{$(1,1)$}, \mbox{$(2,4)$}, \mbox{$(3,9)$}. Is the graph likely to be a straight line or a curve? Explain from the table.

\task The table is \mbox{$(0,5)$}, \mbox{$(2,9)$}, \mbox{$(4,13)$}, \mbox{$(6,17)$}. Write a rule that matches the pattern and then write the point for \mbox{$x=8$}.
\end{tasks}
