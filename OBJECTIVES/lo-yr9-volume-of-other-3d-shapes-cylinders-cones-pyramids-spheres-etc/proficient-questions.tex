{\LARGE \textbf{Volume of other 3D shapes (cylinders, cones, pyramids, spheres, etc) --- Proficient}}
\begin{tasks}(3)
\task A can is a cylinder with radius 4 cm and height 15 cm. Find its volume in terms of \mbox{$\pi$}.
\task A traffic cone has radius 10 cm and height 24 cm. Find its volume in terms of \mbox{$\pi$}.
\task A ball has radius 7 cm. Find its volume in terms of \mbox{$\pi$}.
\task A square pyramid has base side 9 cm and height 14 cm. Find its volume.
\task A rectangular pyramid has base 12 m by 7 m and height 10 m. Find its volume.
\task A cylinder has volume \mbox{$320\pi$} cm\mbox{$^3$} and radius 8 cm. Find its height.
\task A cone has volume \mbox{$75\pi$} m\mbox{$^3$} and radius 5 m. Find its height.
\task A cylinder and a cone have the same radius 6 cm and the same height 12 cm. Find both volumes and compare them.
\task A sphere has volume \mbox{$500\pi/3$} cm\mbox{$^3$}. Find its radius.
\task A square pyramid has volume 392 cm\mbox{$^3$} and height 6 cm. Find the side length of the base.
\task A cylindrical tank has radius 2.5 m and height 3 m. Find its volume correct to 1 decimal place.
\task A hemisphere has radius 9 cm. Find its volume in terms of \mbox{$\pi$}.
\end{tasks}
