{\LARGE \textbf{Volume of other 3D shapes (cylinders, cones, pyramids, spheres, etc) --- Foundation}}
\begin{tasks}(3)
\task A cylinder has radius 3 cm and height 8 cm. Find its volume in terms of \mbox{$\pi$}.
\task A cylinder has radius 5 m and height 4 m. Find its volume in terms of \mbox{$\pi$}.
\task A cone has radius 6 cm and height 9 cm. Find its volume in terms of \mbox{$\pi$}.
\task A cone has radius 4 m and height 12 m. Find its volume in terms of \mbox{$\pi$}.
\task A sphere has radius 3 cm. Find its volume in terms of \mbox{$\pi$}.
\task A sphere has radius 6 m. Find its volume in terms of \mbox{$\pi$}.
\task A square pyramid has base side 10 cm and perpendicular height 9 cm. Find its volume.
\task A square pyramid has base side 12 m and perpendicular height 5 m. Find its volume.
\task A rectangular pyramid has base 8 cm by 5 cm and height 9 cm. Find its volume.
\task A cylinder has volume \mbox{$81\pi$} cm\mbox{$^3$} and height 9 cm. Find the radius.
\task A cone has volume \mbox{$48\pi$} cm\mbox{$^3$} and height 4 cm. Find the radius.
\task A sphere has volume \mbox{$500\pi/3$} cm\mbox{$^3$}. Find the radius.
\task A square pyramid has volume 144 cm\mbox{$^3$} and base side 6 cm. Find the height.
\task A cylinder has radius 7 cm. If the height doubles, what happens to the volume?
\end{tasks}
