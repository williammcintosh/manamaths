\begin{tasks}(3)
\task A student says \mbox{$3+n$} means \textit{three less than \mbox{$n$}}. Are they correct? Explain.
\task A student says \mbox{$2a+5$} and \mbox{$5a+2$} represent the same situation. Are they correct? Explain.
\task Write an expression for the total cost of \mbox{$m$} muffins at 3 dollars each and one drink costing 4 dollars.
\task Write an expression for the perimeter of a rectangle with length \mbox{$x$} and width \mbox{$x+2$}.
\task Write an expression for the number of seats in \mbox{$r$} rows with \mbox{$8$} seats in each row, then remove \mbox{$3$} broken seats.
\task Fill in the blank: \textit{\mbox{$7$} more than \mbox{$p$}} is \rule{2cm}{0.15mm}.
\task Fill in the blank: \textit{half of \mbox{$n$}, then subtract \mbox{$4$}} is \rule{2.6cm}{0.15mm}.
\task Which does not belong: \mbox{$4x$}, \mbox{$x+x+x+x$}, \mbox{$4+x$}, \mbox{$2(2x)$}?
\task Which does not belong: \mbox{$3y+2$}, \mbox{$2+3y$}, \mbox{$y+y+y+2$}, \mbox{$3(y+2)$}?
\task Write an expression for the age of Sam in \mbox{$y$} years if Sam is now \mbox{$12$}.
\task Write an expression for the amount left after spending 5 dollars from \mbox{$x$}.
\task Write an expression for the area of a rectangle with width \mbox{$w$} and length \mbox{$w+6$}.
\task Is \mbox{$5n$} the same as \mbox{$n+5$}? Explain.
\task Complete: if each bag has \mbox{$c$} cookies and there are \mbox{$6$} bags, the total is \rule{2cm}{0.15mm}.
\task Complete: if a taxi fare is 4 dollars plus 2 dollars per kilometre for \mbox{$k$} kilometres, the cost is \rule{2.4cm}{0.15mm}.
\task Write a word description for \mbox{$8-2p$}.
\task Write a word description for \mbox{$3(x+1)$}.
\task Explain why \mbox{$4n$} is a better expression than \mbox{$n+n+n+n$}, even though they are equivalent.
\end{tasks}
