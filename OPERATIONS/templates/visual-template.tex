% Frame builder: 1 image + 3 rows × 3 cards
\newcommand{\visualFrame}[9]{%
  \begin{frame}[t,shrink=30]
  \WorksheetTitle{#1}{#2}
  \vspace{-0.02em}
  % 3 rows
  \visualRow{#3}{#4}{#5}{#6}{#7}{#8}{#9}
  \end{frame}
}

    \begin{column}{0.32\textwidth}\MMProblem{#2}{#3}\end{column}
    \begin{column}{0.32\textwidth}\MMProblem{#4}{#5}\end{column}

}
