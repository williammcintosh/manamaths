\newcommand{\visualRow}[5]{%
  \begin{minipage}[t]{\linewidth}
  \begin{center}
  \includegraphics[width=0.92\linewidth,height=0.40\textheight,keepaspectratio]{#1}
  \end{center}
  \vspace{0.15em}
  \noindent
  \begin{minipage}[t]{0.31\linewidth}\hspace{0.5em}\textbf{#2}\end{minipage}\hfill

  \begin{minipage}[t]{0.31\linewidth}\hspace{0.5em}\textbf{#4}\end{minipage}
  \end{minipage}\par\vspace{0.5em}#5%
}
